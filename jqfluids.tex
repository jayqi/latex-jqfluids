% Dimensionless quantities
\newcommand{\capillary}{\mathrm{Ca}}		% Capillary number
\newcommand{\froude}{\mathrm{Fr}}			% Froude number
\newcommand{\nusselt}{\mathrm{Nu}}		% Nusselt number
\newcommand{\peclet}{\mathrm{Pe}}			% Peclet number
\newcommand{\prandtl}{\mathrm{Pr}}		% Prandtl number
\newcommand{\rayleigh}{\mathrm{Ra}}		% Rayleigh number
\newcommand{\reynolds}{\mathrm{Re}}	 	% Reynolds number
\newcommand{\rossby}{\mathrm{Ro}}		% Rossby number
\newcommand{\schmidt}{\mathrm{Sc}}		% Schmidt number
\newcommand{\strouhal}{\mathrm{St}} 		% Strouhal number
\newcommand{\weber}{\mathrm{We}}		% Weber number
